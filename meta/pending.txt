making pfx file (e.g. from LetsEncrypt):
openssl pkcs12 -export -out certificate.pfx -inkey privateKey.key -in certificate.crt

locally creating pfx:
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout server.key -out server.crt -config req.cnf -extensions 'req_ext'
openssl pkcs12 -export -out server.pfx -inkey server.key -in server.crt

example reqfile:
[req]
default_bits       = 2048
distinguished_name = req_distinguished_name
req_extensions     = req_ext
prompt = yes
[req_distinguished_name]
countryName = Country Name
countryName_default                 = Ameroppa
stateOrProvinceName = State
stateOrProvinceName_default         = High Charity
localityName = City
localityName_default               = Mausoleum of the Arbiter
organizationName = Organization
organizationName_default           = teamRokuro
commonName = Common Name
commonName_default                 = 127.0.0.1
[req_ext]
subjectAltName = @alt_names
[alt_names]
DNS.1   = 127.0.0.1
DNS.2   = localhost
